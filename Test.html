-- ** Datatypes & functions

data Bool : Set where
  true false : Bool

not : Bool → Bool
not true  = false
not false = true

not' : Bool → Bool
not' = λ where
  true  → false
  false → true

module _ (a : Set) where data Maybe : Set where
  nothing : Maybe
  just    : a → Maybe

fromMaybe : a → Maybe a → a
fromMaybe def = λ where
  nothing  → def
  (just a) → a

-- ** Natural numbers

sum : List Nat → Nat
sum []       = 0
sum (x ∷ xs) = x + sum xs

-- ** Polymorphic functions

_++_ : List a → List a → List a
[]       ++ ys = ys
(x ∷ xs) ++ ys = x ∷ (xs ++ ys)

map : (a → b) → List a → List b
map f [] = []
map f (x ∷ xs) = f x ∷ map f xs

plus3 : List Nat → List Nat
plus3 = map (λ n → n + 3)
